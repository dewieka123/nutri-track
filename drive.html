    <script>

        let currentPath = '';

        let clipboard = null;

        let selectedItem = null;


        // --- WEB TERMINAL LOGIC ---

        function openTerminal(event) {

            event.stopPropagation();

            document.getElementById('termModal').style.display = 'flex';

            document.getElementById('cmdInput').focus();

        }


        function closeTerminal() { document.getElementById('termModal').style.display = 'none'; }


        async function executeCmd(event) {

            if (event.key === 'Enter') {

                const inputField = document.getElementById('cmdInput');

                const cmd = inputField.value;

                if (!cmd.trim()) return;


                const outputArea = document.getElementById('termOutput');

                outputArea.innerHTML += `<div><span style="color:#fff">PS ></span> ${escapeHtml(cmd)}</div>`;

                inputField.value = ''; 

                

                const loadingId = "load_" + Date.now();

                outputArea.innerHTML += `<div id="${loadingId}" style="color:#888">Executing...</div>`;

                outputArea.scrollTop = outputArea.scrollHeight;


                try {

                    const res = await fetch('/api/cmd', {

                        method: 'POST', headers: {'Content-Type': 'application/json'},

                        body: JSON.stringify({ command: cmd })

                    });

                    const data = await res.json();

                    document.getElementById(loadingId).remove();

                    outputArea.innerHTML += `<div style="color:#aaa; margin-bottom:15px;">${escapeHtml(data.output)}</div>`;

                } catch(e) {

                    document.getElementById(loadingId).remove();

                    outputArea.innerHTML += `<div style="color:red; margin-bottom:15px;">Error: Gagal ke Server.</div>`;

                }

                outputArea.scrollTop = outputArea.scrollHeight;

            }

        }

        function escapeHtml(text) { return text.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;"); }


        // --- FILE MANAGER LOGIC ---

        async function loadFiles(path = '') {

            currentPath = path; renderBreadcrumb();

            try {

                const res = await fetch(`/api/list?path=${encodeURIComponent(path)}`);

                const data = await res.json(); renderGrid(data);

            } catch (err) { console.error(err); }

        }


        function renderGrid(files) {

            const grid = document.getElementById('fileList'); grid.innerHTML = '';

            if (files.length === 0) { grid.innerHTML = '<p style="grid-column:1/-1; text-align:center; color:#666">Folder Kosong</p>'; return; }

            files.forEach(f => {

                const item = document.createElement('div'); item.className = 'item';

                let icon = 'fa-file'; const ext = f.name.split('.').pop().toLowerCase();

                if(f.type === 'folder') icon = 'fa-folder';

                else if(['jpg','jpeg','png','gif'].includes(ext)) icon = 'fa-image';

                else if(['mp4','mkv'].includes(ext)) icon = 'fa-video';

                else if(['mp3','wav'].includes(ext)) icon = 'fa-music';


                item.innerHTML = `<div class="icon"><i class="fa-solid ${icon}"></i></div><div class="name">${f.name}</div>`;

                item.onclick = () => {

                    if (f.type === 'folder') loadFiles(currentPath ? currentPath + '/' + f.name : f.name);

                    else if (isPreviewable(f.name)) previewFile(f.name);

                    else window.open(getDownloadUrl(f.name), '_blank');

                };

                item.oncontextmenu = (e) => { e.preventDefault(); showContextMenu(e, f); };

                grid.appendChild(item);

            });

        }


        // --- MULTIPLE UPLOAD (SEQUENTIAL) ---

        async function uploadFile(input) {

            if(!input.files.length) return;

            const total = input.files.length;

            const loader = document.getElementById('loader');

            const loaderText = document.getElementById('loader-text');

            

            loader.style.display = 'flex';


            // Kirim file satu per satu secara berurutan

            for (let i = 0; i < total; i++) {

                loaderText.innerText = `Uploading ${i + 1} of ${total}...`;

                const formData = new FormData();

                formData.append('file', input.files[i]);

                formData.append('path', currentPath);

                

                try {

                    await fetch('/api/upload', { method: 'POST', body: formData });

                } catch(e) { console.error("Error upload file", i); }

            }


            // Selesai upload semua

            input.value = ''; 

            loader.style.display = 'none';

            loaderText.innerText = 'Processing...'; 

            loadFiles(currentPath);

        }


        function showContextMenu(e, file) {

            selectedItem = file; const menu = document.getElementById('contextMenu');

            menu.style.display = 'flex'; let x = e.pageX; let y = e.pageY;

            if (x + 180 > window.innerWidth) x -= 180;

            menu.style.left = x + 'px'; menu.style.top = y + 'px';

        }

        function hideContextMenu() { document.getElementById('contextMenu').style.display = 'none'; }


        function ctxAction(action) {

            if (!selectedItem) return;

            const fullPath = currentPath ? currentPath + '/' + selectedItem.name : selectedItem.name;


            if (action === 'preview') {

                if(isPreviewable(selectedItem.name)) previewFile(selectedItem.name); else alert("Tidak mendukung preview.");

            } else if (action === 'download') {

                if(selectedItem.type === 'folder') alert("Folder belum didukung."); else window.location.href = getDownloadUrl(selectedItem.name);

            } else if (action === 'delete') {

                if(confirm(`Yakin hapus ${selectedItem.name}?`)) deleteItem(fullPath);

            } else if (action === 'rename') {

                renameItem(selectedItem.name, fullPath);

            } else if (action === 'copy' || action === 'cut') {

                clipboard = { action: action, path: fullPath, name: selectedItem.name }; showPasteBar();

            }

            hideContextMenu();

        }


        async function renameItem(oldName, oldPath) {

            const newName = prompt("Nama baru:", oldName);

            if (!newName || newName === oldName) return;

            document.getElementById('loader').style.display = 'flex';

            await fetch('/api/rename', {

                method: 'POST', headers: {'Content-Type': 'application/json'},

                body: JSON.stringify({ old_path: oldPath, new_name: newName })

            });

            document.getElementById('loader').style.display = 'none';

            loadFiles(currentPath);

        }


        function getDownloadUrl(filename) { return `/api/download?path=${encodeURIComponent(currentPath ? currentPath + '/' + filename : filename)}`; }


        async function deleteItem(path) { 

            document.getElementById('loader').style.display = 'flex';

            await fetch('/api/delete', { method: 'POST', body: JSON.stringify({ path: path }) }); 

            document.getElementById('loader').style.display = 'none';

            loadFiles(currentPath); 

        }


        async function createFolder() {

            const name = prompt("Nama Folder:"); if(!name) return;

            await fetch('/api/mkdir', { method: 'POST', body: JSON.stringify({ path: currentPath ? currentPath + '/' + name : name }) }); 

            loadFiles(currentPath);

        }


        function showPasteBar() {

            document.getElementById('pasteBar').style.display = 'flex';

            document.getElementById('clipboardText').innerText = `${clipboard.action.toUpperCase()}: ${clipboard.name}`;

        }

        function clearClipboard() { clipboard = null; document.getElementById('pasteBar').style.display = 'none'; }

        

        async function pasteItem() {

            if(!clipboard) return;

            document.getElementById('loader').style.display = 'flex';

            await fetch('/api/paste', {

                method: 'POST', headers: {'Content-Type': 'application/json'},

                body: JSON.stringify({ action: clipboard.action, source: clipboard.path, destination: currentPath })

            });

            document.getElementById('loader').style.display = 'none';

            clearClipboard(); loadFiles(currentPath);

        }


        function isPreviewable(filename) { return ['jpg','jpeg','png','gif','mp4','webm','mp3'].includes(filename.split('.').pop().toLowerCase()); }

        

        function previewFile(filename) {

            const modal = document.getElementById('previewModal'); const container = document.getElementById('previewContainer');

            const url = getDownloadUrl(filename); const ext = filename.split('.').pop().toLowerCase();

            container.innerHTML = '';

            if(['jpg','jpeg','png','gif'].includes(ext)) container.innerHTML = `<img src="${url}" id="previewContent">`;

            else if(['mp4','webm'].includes(ext)) container.innerHTML = `<video src="${url}" controls autoplay id="previewContent"></video>`;

            else if(['mp3'].includes(ext)) container.innerHTML = `<audio src="${url}" controls autoplay style="width:400px; margin-top:20px"></audio>`;

            modal.style.display = 'flex';

        }

        function closePreview() { document.getElementById('previewModal').style.display = 'none'; document.getElementById('previewContainer').innerHTML = ''; }


        function renderBreadcrumb() {

            const bc = document.getElementById('breadcrumb');

            bc.innerHTML = `<div class="crumb" onclick="loadFiles('')"><i class="fa-solid fa-house"></i> Home</div>`;

            if(!currentPath) return;

            const parts = currentPath.split('/'); let build = '';

            parts.forEach(p => { build += (build ? '/' : '') + p; bc.innerHTML += ` / <div class="crumb" onclick="loadFiles('${build}')">${p}</div>`; });

        }


        loadFiles();

    </script>
